\roadmapSubSection{5.1}{Week-by-Week Execution Plan}{32 Weeks}

{\small\itshape\color{arligray} Assumes 4--6 hours/day focused work. Adjust linearly for your available hours.}

\roadmapHead{Phase 1 --- Foundations (Weeks 1--8)}

\timelineitem{Weeks 1--2}{Safety, ESD, Workspace, Tools}
  {Buy all tools. Set up ESD-safe workspace. Study battery safety. Do not touch live boards until workspace complete.}

\timelineitem{Weeks 3--6}{Electronics Fundamentals}
  {Ohm's Law through power sequencing. Build practice circuits. Measure every component type physically with LCR + DMM.}

\timelineitem{Weeks 7--8}{Soldering and Micro-Soldering}
  {Daily practice on dead boards. 0402/0805/QFN/BGA. First jumper wire repair. Hot air control.}

\milestoneitem{Week 8 Milestone}{}
  {Can solder 0402 reliably. Can read schematic and trace a power rail. Workspace fully operational.}

\roadmapHead{Phase 2 --- Mobile Repair Income (Weeks 9--22)}

\timelineitem{Weeks 9--12}{Common Mobile Repairs}
  {30+ screen, battery, port, camera, speaker replacements. Diagnostic sequence mastered on live boards.}

\milestoneitem{Week 12 --- First Income}{}
  {Start accepting repair jobs. Screen, battery, port replacements only. Build client base and reputation.}

\timelineitem{Weeks 13--17}{Android Board-Level}
  {Qualcomm EDL, Samsung Odin, MTK SP Flash, UFI Box eMMC, UART boot log, first current-draw diagnosis.}

\timelineitem{Weeks 18--22}{iPhone Board-Level}
  {Charging IC, PMIC, NAND/chain-of-trust concepts, True Tone via JCID, Face ID rules, DFU/checkm8.}

\milestoneitem{Week 22 Milestone}{}
  {Board-level mobile repair operational. Rate: \$80--200/repair depending on fault. Steady weekly income.}

\roadmapHead{Phase 3 --- Laptop Repair (Weeks 23--28)}

\timelineitem{Weeks 23--25}{Laptop Mechanical and Architecture}
  {DC jack, keyboard, screen, SSD, thermal paste. VRM theory. BIOS chip read/write with CH341A.}

\timelineitem{Weeks 26--28}{Laptop Board-Level}
  {No-power diagnosis, VRM phase fault, charging IC, RAM fault, liquid damage protocol.}

\milestoneitem{Week 28 Milestone}{}
  {Laptop repair operational. Second reliable income stream. Adds 30--50\% more monthly revenue.}

\roadmapHead{Phase 4 --- PC and Advanced Tools (Weeks 29--32+)}

\timelineitem{Weeks 29--32}{PC Hardware, OS, Peripherals}
  {Motherboard diagnosis, capacitor replacement, PSU test, Windows/Linux/macOS install. Monitor repair basics.}

\timelineitem{Weeks 28--32 (Parallel)}{Oscilloscope and Thermal Imaging}
  {Acquire scope and thermal camera. Begin power rail ripple measurements and thermal fault isolation.}

\milestoneitem{Week 32 Milestone}{}
  {Full repair stack operational. Mobile + Laptop + PC. Oscilloscope-guided diagnosis reduces time per job by 50\%.}
