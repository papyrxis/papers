\roadmapSubSection{1.1}{Common Mobile Diagnostics}{4 Weeks}

{\small\itshape\color{arligray} Learn this once. Applies to Android and iPhone equally. The diagnostic method never changes.}

\roadmapHead{Diagnostic Methodology --- Always Follow This Exact Sequence}
\checkitem{Step 1 --- Visual inspection under microscope: corrosion, liquid residue, charred ICs, cracked joints}{}
\checkitem{Step 2 --- Diode mode test (board unpowered): red probe to ground, black to power rails}{Normal: 0.4--0.7V | Short: near 0V | Open: OL | Compare to known-good board}
\checkitem{Step 3 --- DC power supply, current limit 0.5A, observe at power button press}{0mA after button: open fuse/dead PMIC | 1--5A instant: direct short on VBAT | 50--150mA stuck: RAM init fail}

\roadmapHead{Short Circuit Diagnosis}
\checkitem{Confirm shorted rail via diode mode (reads near 0 ohms)}{}
\checkitem{Use freeze spray or thermal camera to find shorted component (gets warm first)}{}
\checkitem{Identify and remove the shorted component (usually a capacitor or IC)}{}
\checkitem{Verify short cleared by retesting diode mode --- must read normal before proceeding}{}
\checkitem{Replace removed component and retest}{}

\roadmapHead{Common Repairs --- 30+ Completions Before iPhone Board-Level}
\checkitem{Screen replacement --- disconnect battery first, every time, no exceptions}{Warm adhesive with separator plate at 80°C for 2--3 min before lifting display}
\checkitem{Battery replacement --- discharge below 20\%, never puncture, dispose swollen cells safely}{}
\checkitem{Charging port replacement --- identify port type, clean pads, rework or replace}{}
\checkitem{Power button and volume flex cable replacement}{}
\checkitem{Front and rear camera module replacement}{}
\checkitem{Earpiece and loudspeaker replacement}{}
\checkitem{Microphone replacement or jumper bypass when traces broken}{}
\checkitem{Back glass removal and adhesive re-application}{}

\roadmapHead{Reading Mobile Datasheets and Schematics}
\checkitem{Locate and download schematic via MaAnt or S-Finder}{}
\checkitem{Identify all main ICs: PMIC, charge IC, CPU/AP, baseband, WiFi, audio, display}{}
\checkitem{Trace charging path: USB connector $\to$ charge IC $\to$ battery --- know every node}{}
\checkitem{Trace display path: AP $\to$ display IC $\to$ panel connector}{}
\checkitem{Trace audio path: AP $\to$ audio codec $\to$ speaker and microphone}{}
\checkitem{Trace RF path: AP $\to$ baseband $\to$ antenna switches $\to$ antennas}{}
\checkitem{For every IC you repair around: read its datasheet --- first 10 pages minimum}{}

\checkitem{Repair charging fault: trace VBUS path, test charge IC output rails}{}

\roadmapHead{eMMC and UFS Repair with UFI Box}

\roadmapHead{UART Boot Log Reading on Android}

\roadmapHead{iPhone Architecture --- Understand Before Touching}
\checkitem{Apple Silicon A-series: CPU + GPU + Neural Engine + Secure Enclave in one die}{}
\checkitem{NAND pairing: iPhone NAND and CPU are cryptographically paired --- NAND swap = dead device}{}
\checkitem{Baseband certificate: IMEI tied to baseband at factory level, cannot be software-changed}{}
\checkitem{Face ID pairing: dot projector paired to logic board --- not field-replaceable}{}
\checkitem{True Tone pairing: display calibration tied to logic board --- use JCID to transfer}{}
\checkitem{Chain of trust: SecureROM $\to$ LLB $\to$ iBoot $\to$ kernel $\to$ SpringBoard --- each stage verified}{}

\roadmapHead{iOS Boot Modes}
\checkitem{Understand normal boot vs recovery mode vs DFU mode --- know how to trigger each}{}
\checkitem{DFU mode: bypasses iBoot, used for deep restore}{}
\checkitem{Recovery mode: iBoot waits for iTunes/Finder restore --- shows iTunes logo}{}
\checkitem{SHSH blobs: version-specific signing tickets --- why iOS downgrades are blocked}{}

\roadmapHead{iPhone Board-Level Repairs}
\checkitem{Charging fault: trace TRISTAR/HYDRA/ACE3 IC, test PP5V0\_USB rail}{}
\checkitem{No power: test PP\_BATT\_VCC rail, locate main power button IC (Hazel/Sage)}{}
\checkitem{Bootloop: check NAND communication bus, AP-to-NAND SPI lines, confirm boot log via UART}{}
\checkitem{Audio fault: locate audio IC (338S), check AVDD rail, reball if needed}{}
\checkitem{Baseband no service: check baseband PMU rails, RF ICs, antenna switch control lines}{}
\checkitem{WiFi fault: locate WiFi/BT combo IC, check WLAN\_VCC rail and antenna connection}{}
\checkitem{Face ID repair: check dot projector flex connector, verify pairing via JCID scanner}{}
\checkitem{Screen replacement with True Tone calibration transfer via JCID}{}
\checkitem{Battery swap with health percentage reset via JCID}{}
