\roadmapSection{11}{INCOME ROADMAP AND BUSINESS PLAN}{}

{\small\itshape\color{arligray} Repair income funds everything else. This track never stops. Never let income slip for hobby tracks.}

% ── 11.1 Income Roadmap ───────────────────────────────────────────────────────
\roadmapSubSection{11.1}{Income Roadmap --- Month by Month}{}

\roadmapHead{Months 1--3: Skill Building, No Clients}
\checkitem{Practice on dead boards only. Do not take customer devices}{}
\checkitem{Complete 20+ screen and battery replacements on practice phones}{}
\checkitem{Complete 10+ board-level diagnoses on dead Android boards}{}
\checkitem{Research repair hubs in your city: locations, rent cost, foot traffic}{}

\roadmapHead{Months 4--6: First Income}
\checkitem{Find Miz-e Kar (table rental) inside established repair hub --- negotiate fixed fee or profit share}{}
\checkitem{Accept only basic repairs: screen, battery, charging port, software flash}{}
\checkitem{Price slightly below market rate while building reputation}{}
\checkitem{Document every repair with photos --- build portfolio and customer trust}{}
\checkitem{Track: repairs/month, revenue/month, most common fault type}{}
\checkitem{Target: 20 repairs/month by month 5, 35 repairs/month by month 6}{}

\roadmapHead{Months 7--9: Expand Services and Income}
\checkitem{Add Android board-level repairs to service list (30+ successful board repairs done)}{}
\checkitem{Set board-level pricing: 300--600k Toman labor per job}{}
\checkitem{Add data recovery: 500k--1M Toman --- buy original UFI Box first}{}
\checkitem{Build referral network: other technicians send you board-level, you share 10--15\%}{}
\checkitem{Target: 50--70 repairs/month}{}

\roadmapHead{Months 10--12: Formalize}
\checkitem{Apply for Mobile Repair Skill Certificate Class 1 (TVTO)}{}
\checkitem{Apply for business license via national license portal}{}
\checkitem{Open dedicated business bank account}{}
\checkitem{Add iPhone board-level after certification --- highest margin service}{}
\checkitem{Target: 100+ repairs/month, income covering all living expenses}{}

\roadmapHead{Months 13--18: Scale and Specialize}
\checkitem{Add laptop and PC board-level repair --- second income stream}{}
\checkitem{Add premium ARLIZ consulting for engineering students}{}
\checkitem{Consider online repair consulting: diagnose remotely via photos and video}{}
\checkitem{Build GitHub portfolio: ARLIZ chapters, Talken hardware, firmware code}{}
\checkitem{Target: stable income that can fund Estonia relocation}{}

\roadmapHead{Customer Acquisition}
\checkitem{Location: high foot-traffic mobile hub gives immediate walk-in customer access}{}
\checkitem{Word of mouth: every happy customer = 2--3 referrals. Every unhappy = 10 negative}{}
\checkitem{Online presence: Google Maps listing, before/after Instagram posts of repairs}{}
\checkitem{Warranty: offer it clearly. Most customers pay more for confidence}{}

\roadmapHead{Legal Requirements (Iran)}
\checkitem{Mobile Repair Skill Certificate Class 2 $\to$ Class 1 (TVTO)}{}
\checkitem{Business license via national license portal}{}
\checkitem{Trade union registration: Union of Telecommunications Devices}{}
\checkitem{Military service documentation (if applicable): completion card, exemption, or student status}{}
\checkitem{Minimum commercial space: 6--8 square meters with active utilities}{}
