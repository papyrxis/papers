\roadmapSection{10}{ENGLISH AND TOEFL PREPARATION}{Every Morning $\cdot$ 90 Min}

{\small\itshape\color{arligray} English is the gate to Estonia and international career. Non-negotiable daily practice.}

% ── 10.1 Daily English Practice ───────────────────────────────────────────────
\roadmapSubSection{10.1}{Daily English Practice}{}

\checkitem{Anki flashcard deck: 100 new vocabulary words per day, review old cards first}{}
\checkitem{One technical article per day: EEVblog post, iFixit teardown, IETF RFC, IEEE abstract}{}
\checkitem{Add new unknown words from article directly to Anki deck --- same session}{}
\checkitem{One paragraph written in English per day: summarize what you learned technically}{}
\checkitem{Record yourself speaking English for 2 minutes: describe today's repair or circuit}{}

% ── 10.2 TOEFL Preparation ────────────────────────────────────────────────────
\roadmapSubSection{10.2}{TOEFL Preparation}{}

\roadmapHead{Reading Section}
\checkitem{Practice 3 academic passages per week --- time yourself (20 minutes per passage)}{}
\checkitem{Focus on: inference questions, vocabulary in context, rhetorical purpose}{}
\checkitem{Read 2 academic papers per month from IEEE or ACM --- build academic vocabulary}{}

\roadmapHead{Listening Section}
\checkitem{Practice 2 lecture sets + 2 conversation sets per week}{}
\checkitem{Take notes in English while listening --- summarize main idea and 3 supporting points}{}
\checkitem{Listen to MIT OpenCourseWare lectures relevant to your study (circuits, CS)}{}

\roadmapHead{Speaking Section}
\checkitem{Task 1 (independent): pick a topic, speak 45 seconds, record and evaluate}{}
\checkitem{Tasks 2--4 (integrated): read + listen + speak --- practice note-taking under time pressure}{}
\checkitem{Common errors to eliminate: article usage (a/an/the), verb tense consistency, prepositions}{}

\roadmapHead{Writing Section}
\checkitem{Integrated task: read passage, listen to lecture, write 150--225 word response}{}
\checkitem{Independent task: write 300+ word essay on a given topic --- practice weekly}{}
\checkitem{Academic writing style: thesis statement, body paragraphs with evidence, conclusion}{}

\roadmapHead{TOEFL Milestones}
\checkitem{Month 4: TOEFL practice test --- target 70+ score}{}
\checkitem{Month 8: TOEFL practice test --- target 85+ score}{}
\checkitem{Month 12: TOEFL practice test --- target 95+ score}{}
\checkitem{Month 15--16: Register for official TOEFL --- target 100+ (required for Estonia programs)}{}
