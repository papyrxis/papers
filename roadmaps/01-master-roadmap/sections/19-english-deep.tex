% ── 17.2 Writing --- Technical and Academic ────────────────────────────────────
\roadmapSubSection{17.2}{Technical Writing and Academic Style}{Daily --- 30 Minutes}

\roadmapHead{Daily Writing Practice}
\checkitem{Write one paragraph per day: describe what you repaired, learned, or built --- in English}{}
\checkitem{Keep an engineering notebook in English: date, title, what I did, what I observed, what I concluded}{}
\checkitem{ARLIZ chapters are English writing practice. Write the chapter, then improve the English}{}
\checkitem{GitHub commit messages: always in English. Present tense. ``Add I2C driver for BMP280''}{}
\checkitem{GitHub README files: write in English. Explain what the project does, how to build, how to use}{}

\roadmapHead{Academic Writing Structure (TOEFL Independent Task)}
\checkitem{Paragraph structure: topic sentence → evidence/example → analysis → transition}{}
\checkitem{Thesis statement: clear position in first paragraph. Specific, arguable, not a fact}{}
\checkitem{Cohesion devices: however, therefore, consequently, in addition, specifically, to illustrate}{}
\checkitem{Avoid: very, good, bad, thing, got. Replace with: significantly, effective/efficient, detrimental, component, obtained}{}
\checkitem{Write 300+ word essay every weekend on given topic. Time yourself to 30 minutes}{}
\checkitem{Self-evaluate: check for article errors (a/an/the), verb tense consistency, subject-verb agreement}{}

\roadmapHead{Common Grammar Errors to Eliminate}
\checkitem{Articles: ``a'' before consonant sound, ``an'' before vowel sound, ``the'' for specific known entities}{}
\checkitem{Countable vs uncountable: ``information'' is uncountable (no ``informations''), ``data'' same}{}
\checkitem{Verb tenses: simple past for completed events, present perfect for recent/relevant past}{}
\checkitem{Relative clauses: ``which'' for non-restrictive, ``that'' for restrictive. ``The chip, which is...'' vs ``The chip that handles...'' }{}
\checkitem{Prepositions: ``depend on'', ``consist of'', ``result in'', ``responsible for'' --- memorize collocations}{}
\checkitem{Passive voice: correct use in technical writing. ``The signal is filtered by the RC circuit''}{}

\checkitem{Sentence stress: content words stressed, function words reduced. ``I WENT to the STORE''}{}
\checkitem{Intonation: falling for statements/commands, rising for yes/no questions}{}
\checkitem{TOEFL Speaking Task 1: 15s prep, 45s speak. Practice 5 topics per week}{}
\checkitem{TOEFL Speaking Task 2--4: integrated tasks. Read + listen + synthesize in speech}{}

% ── 17.4 TOEFL Preparation --- Section by Section ────────────────────────────
\roadmapSubSection{17.4}{TOEFL iBT Preparation --- 100+ Score}{Weekly Blocks}

\roadmapHead{TOEFL Reading Section (target: 27+)}
\checkitem{3--4 passages, 10 questions each, 54--72 minutes. Topics: science, social science, humanities}{}
\checkitem{Question types: factual information, negative factual, inference, vocabulary, rhetorical purpose, insert sentence, prose summary}{}
\checkitem{Strategy: read introduction + first sentence of each paragraph → answer questions → go back for details}{}
\checkitem{Insert sentence questions: look for referential links (this/these/such) and logical transitions}{}
\checkitem{Prose summary: 6 choices, pick 3 major points. Eliminate detail-only options}{}
\checkitem{Practice source: Official TOEFL Practice Online (ETS), TPO tests on TST Prep, Magoosh TOEFL}{}
\checkitem{Time pressure: 18 minutes per passage. Practice timed from week 1}{}

\roadmapHead{TOEFL Listening Section (target: 26+)}
\checkitem{4--6 lectures + 2--3 conversations. Each 3--5 minutes. No subtitles, note-taking allowed}{}
\checkitem{Note-taking template: Main topic / Purpose. Key points 1, 2, 3. Examples. Conclusion}{}
\checkitem{Lecture types: academic monologue with student questions. Topics overlap with reading}{}
\checkitem{Conversation types: office hours, service encounter. Listen for purpose and attitude}{}
\checkitem{Practice listening to Crash Course (YouTube) without subtitles, take structured notes}{}
\checkitem{Inference questions: what does the speaker imply? What is the attitude? Read tone not just words}{}

\roadmapHead{TOEFL Writing Section (target: 24+)}
\checkitem{Task 1 (Integrated): read 250-word passage (3 min) → listen to lecture → write 150--225 words}{}
\checkitem{Integrated strategy: the lecture always contradicts the reading. Organize by 3 lecture points}{}
\checkitem{Template: ``The reading and the lecture are about [topic]. While the reading argues [X], the lecture refutes this by [Y].''}{}
\checkitem{Task 2 (Academic Discussion): 10-minute discussion board style. 150+ words. New in 2023}{}
\checkitem{Academic Discussion: read two classmate posts, add your view with reasoning and support}{}
\checkitem{Grading criteria: relevance, development, language use, grammar accuracy}{}

\roadmapHead{TOEFL Score Milestones and Test Registration}
\checkitem{Month 4: complete TPO 1 under real conditions --- target 70+}{}
\checkitem{Month 8: complete TPO 5 under real conditions --- target 85+}{}
\checkitem{Month 12: complete TPO 10 under real conditions --- target 95+}{}
\checkitem{Month 14: register official TOEFL iBT at ets.org. Fee \$245 USD (check current price)}{}
\checkitem{Month 15: take official exam. Target 100+. Required for TalTech and most Estonian programs}{}
\checkitem{TOEFL score validity: 2 years from test date. Plan registration to arrive before visa application}{}
\checkitem{Test center in Iran: available in Tehran. Confirm availability and register 2--3 months ahead}{}

% ── 17.5 Technical English --- Engineering Communication ─────────────────────
\roadmapSubSection{17.5}{Technical English for Engineering Career}{Ongoing}

\roadmapHead{Writing for Technical Documentation}
\checkitem{User manual style: numbered steps, imperative mood. ``Connect the power supply to the DC jack.''}{}
\checkitem{Datasheet style: tables, conditions, min/typ/max values. Learn to read and write this format}{}
\checkitem{Bug report style: steps to reproduce, expected behavior, actual behavior, environment}{}
\checkitem{Technical email: subject line is summary. First sentence is purpose. Short paragraphs}{}
\checkitem{Stack Overflow / GitHub issue style: minimal reproducible example, exact error message}{}
\checkitem{ARLIZ book style: explain to a technical reader who is one level below your current knowledge}{}

\roadmapHead{Technical Vocabulary by Domain}
\checkitem{Electronics: impedance, attenuation, rectification, modulation, transient, propagation delay}{}
\checkitem{Embedded: preemption, interrupt latency, DMA, memory-mapped I/O, watchdog, bootloader}{}
\checkitem{Repair: desoldering, rework, reflow, continuity, short-circuit, open-circuit, diode mode}{}
\checkitem{Programming: pointer arithmetic, heap allocation, race condition, mutex, semaphore, overhead}{}
\checkitem{RF/Wireless: gain, noise floor, SNR, bandwidth, RSSI, antenna polarization, multipath}{}
\checkitem{Learn each term: definition, sentence context, how it sounds in speech, abbreviation if any}{}
