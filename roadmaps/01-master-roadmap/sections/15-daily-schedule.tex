\roadmapSubSection{13.1}{Daily Structure}{Peak Deep Work = 4--6 Hours}

{\small\itshape\color{arligray} 15 hours of sustained deep work is impossible. Plan around peak cognitive hours.}

\vspace{0.2cm}

\begin{tabular}{>{\bfseries\color{arliznavy}}p{2.5cm} >{\bfseries\color{arlizblue}}p{1.8cm} p{10cm}}
\rowcolor{arliznavy}
\color{white}\textbf{Block} & \color{white}\textbf{Duration} & \color{white}\textbf{Activity} \\
Morning     & 90 min   & English $\cdot$ 100 Anki cards $\cdot$ 1 technical article $\cdot$ 1 paragraph written \\
\rowcolor{arlizlgray}
Pre-noon    & 3 hours  & Main Work Block A --- deepest block. Phone off. No AI. No extra tabs \\
Noon        & 30 min   & Real rest. Walk. No screen \\
\rowcolor{arlizlgray}
Afternoon   & 2 hours  & Main Work Block B --- continue current phase. Read AND do, not read only \\
Evening     & 90 min   & ARLIZ --- write what you learned. One note + one chapter paragraph \\
\rowcolor{arlizlgray}
Late evening & 30 min  & Mathematics. Mastering Fundamentals. No exceptions, even if tired \\
Night       & 30 min   & Journal --- what I learned, what broke, why, what tomorrow holds \\
\end{tabular}

\vspace{0.3cm}

\begin{northstarbox}
\textbf{If only 4 hours available:}\\
\hspace{4pt}$\to$ Block A first\quad$\to$ ARLIZ second\quad$\to$ English third\\
\hspace{4pt}$\to$ Protect the morning block above all\\[6pt]
\textbf{AI Rule:} use AI to check work you already did --- not to replace thinking.\\
Asking AI what a datasheet says = building dependency, not skill.
\end{northstarbox}
