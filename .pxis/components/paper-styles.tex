% ══════════════════════════════════════════════════════════════════════════════
% ACADEMIC — Formal research paper
% Times body, tight margins, black/dark links, theorem environments
% ══════════════════════════════════════════════════════════════════════════════
\@namedef{paperstyle@academic}{%
  \@papergeometry{%
    a4paper,
    top=2.5cm,
    bottom=2.5cm,
    left=2.8cm,
    right=2.8cm,
    headheight=14pt,
    headsep=12pt,
    footskip=28pt
  }%
  \usepackage{times}%
  \usepackage{mathptmx}%
  \usepackage[T1]{fontenc}%
  \setlength{\parindent}{1em}%
  \setlength{\parskip}{0pt}%
  %
  % Override link colors to academic black
  \@ifpackageloaded{hyperref}{%
    \hypersetup{linkcolor=black, citecolor=black, urlcolor=blue}%
  }{}%
  %
  % Theorem environments (amsthm)
  \@ifpackageloaded{amsthm}{}{%
    \usepackage{amsthm}%
  }%
  \theoremstyle{plain}%
  \newtheorem{theorem}{Theorem}[section]%
  \newtheorem{lemma}[theorem]{Lemma}%
  \newtheorem{proposition}[theorem]{Proposition}%
  \newtheorem{corollary}[theorem]{Corollary}%
  \theoremstyle{definition}%
  \newtheorem{definition}[theorem]{Definition}%
  \newtheorem{example}[theorem]{Example}%
  \theoremstyle{remark}%
  \newtheorem{remark}[theorem]{Remark}%
  \newtheorem{note}[theorem]{Note}%
  %
  % Compact section headers
  \usepackage{fancyhdr}%
  \usepackage{lastpage}%
  \pagestyle{fancy}%
  \fancyhf{}%
  \renewcommand{\headrulewidth}{0.3pt}%
  \renewcommand{\footrulewidth}{0pt}%
  \fancyhead[L]{\small\itshape\nouppercase{\leftmark}}%
  \fancyhead[R]{\small\thepage}%
  %
  \def\@paperstyle{academic}%
}

% ══════════════════════════════════════════════════════════════════════════════
% IEEE — Two-column IEEE conference format
% Times, tight spacing, numeric citations, no colored links
% ══════════════════════════════════════════════════════════════════════════════
\@namedef{paperstyle@ieee}{%
  \@papergeometry{%
    a4paper,
    twocolumn,
    columnsep=0.5cm,
    top=2cm,
    bottom=2cm,
    left=1.6cm,
    right=1.6cm
  }%
  \usepackage{times}%
  \usepackage{mathptmx}%
  \usepackage[T1]{fontenc}%
  \setlength{\parindent}{1em}%
  \setlength{\parskip}{0pt}%
  %
  \@ifpackageloaded{hyperref}{%
    \hypersetup{colorlinks=false, pdfborder={0 0 0}}%
  }{}%
  %
  % IEEE-style section headings
  \makeatletter
  \renewcommand{\section}{%
    \@startsection{section}{1}{\z@}%
      {-2.5ex \@plus -0.8ex \@minus -.2ex}%
      {1.5ex \@plus .2ex}%
      {\normalfont\normalsize\bfseries\MakeUppercase}%
  }%
  \renewcommand{\subsection}{%
    \@startsection{subsection}{2}{\z@}%
      {-2ex \@plus -0.5ex \@minus -.2ex}%
      {1ex \@plus .2ex}%
      {\normalfont\small\bfseries\itshape}%
  }%
  \makeatother
  %
  % IEEE keywords environment
  \newenvironment{IEEEkeywords}{%
    \par\noindent\textbf{\textit{Index Terms---}}\ignorespaces
  }{%
    \par
  }%
  %
  \def\@paperstyle{ieee}%
}
