% ── Geometry helper ───────────────────────────────────────────────────────────
\newcommand{\@papergeometry}[1]{%
  \@ifpackageloaded{geometry}{%
    \newgeometry{#1}%
  }{%
    \usepackage[#1]{geometry}%
  }%
}

% ── Public API ────────────────────────────────────────────────────────────────
\newcommand{\applyPaperStyle}[1]{%
  \@nameuse{paperstyle@#1}%
}

% ══════════════════════════════════════════════════════════════════════════════
% PERSONAL — personal essay / technical note
% Charter body, generous margins, branded footer with author + page + year
% ══════════════════════════════════════════════════════════════════════════════
\@namedef{paperstyle@personal}{%
  \@papergeometry{%
    a4paper,
    top=3cm,
    bottom=3cm,
    left=3.2cm,
    right=3.2cm,
    headheight=14pt,
    headsep=18pt,
    footskip=30pt
  }%
  \usepackage{charter}%
  \usepackage[T1]{fontenc}%
  \setlength{\parindent}{1.4em}%
  \setlength{\parskip}{0.5ex plus 0.2ex minus 0.1ex}%
  \usepackage{setspace}%
  \setstretch{1.2}%
  %
  \usepackage{fancyhdr}%
  \usepackage{lastpage}%
  \pagestyle{fancy}%
  \fancyhf{}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0.4pt}%
  \fancyfoot[L]{\scriptsize\textcolor{textsecondary}{\textit{\FooterAuthorName}}}%
  \fancyfoot[C]{\scriptsize\textcolor{textsecondary}{\thepage\ / \pageref{LastPage}}}%
  \fancyfoot[R]{\scriptsize\textcolor{textsecondary}{\textit{\the\year}}}%
  \renewcommand{\footrule}{%
    \hbox to\headwidth{\color{border}\leaders\hrule height 0.4pt\hfill}%
  }%
  %
  \def\@paperstyle{personal}%
}

% ══════════════════════════════════════════════════════════════════════════════
% IEEE — two-column IEEE conference format
% Times, tight spacing, no colored links, uppercase section headings
% Author shown in left footer every page
% ══════════════════════════════════════════════════════════════════════════════
\@namedef{paperstyle@ieee}{%
  \@papergeometry{%
    a4paper,
    twocolumn,
    columnsep=0.5cm,
    top=2cm,
    bottom=2cm,
    left=1.6cm,
    right=1.6cm,
    footskip=22pt
  }%
  \usepackage{times}%
  \usepackage{mathptmx}%
  \usepackage[T1]{fontenc}%
  \setlength{\parindent}{1em}%
  \setlength{\parskip}{0pt}%
  %
  \@ifpackageloaded{hyperref}{%
    \hypersetup{colorlinks=false, pdfborder={0 0 0}}%
  }{}%
  %
  % IEEE-style uppercase section headings
  \renewcommand{\section}{%
    \@startsection{section}{1}{\z@}%
      {-2.5ex \@plus -0.8ex \@minus -.2ex}%
      {1.5ex \@plus .2ex}%
      {\normalfont\normalsize\bfseries\MakeUppercase}%
  }%
  \renewcommand{\subsection}{%
    \@startsection{subsection}{2}{\z@}%
      {-2ex \@plus -0.5ex \@minus -.2ex}%
      {1ex \@plus .2ex}%
      {\normalfont\small\bfseries\itshape}%
  }%
  %
  % IEEE keywords environment
  \newenvironment{IEEEkeywords}{%
    \par\noindent\textbf{\textit{Index Terms---}}\ignorespaces
  }{%
    \par
  }%
  %
  % Minimal footer: author left, page right
  \usepackage{fancyhdr}%
  \usepackage{lastpage}%
  \pagestyle{fancy}%
  \fancyhf{}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0pt}%
  \fancyfoot[L]{\scriptsize\textit{\FooterAuthorName}}%
  \fancyfoot[R]{\scriptsize\thepage}%
  %
  \def\@paperstyle{ieee}%
}
