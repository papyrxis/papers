% Called once in each resume.tex preamble to wire up the theme.

% LTR: applies \titleformat{\section}
\newcommand{\ThemeApplySectionFormat}{%
  \titleformat{\section}%
    {\normalsize\bfseries}{}{0pt}{}%
    [\vspace{-5pt}\color{ResumeAccent}\rule{\linewidth}{\ThemeSectionRuleThickness}]%
  \titlespacing*{\section}{0pt}{\ThemeSectionSpaceAbove}{\ThemeSectionSpaceBelow}%
}

% LTR: applies global itemize defaults
\newcommand{\ThemeApplyListStyle}{%
  \setlist[itemize]{%
    nosep,%
    leftmargin=\ThemeItemLeftMargin,%
    itemsep=\ThemeItemSep,%
    topsep=\ThemeItemTopSep,%
    parsep=\ThemeItemParsep%
  }%
}

% All: hyperref setup — \ThemeApplyHyperref{title}{author}{keywords}
\newcommand{\ThemeApplyHyperref}[3]{%
  \hypersetup{%
    colorlinks=true,%
    urlcolor=\ThemeHyperlinkColor,%
    pdfborder={0 0 \ThemeLinkBorderWidth},%
    pdftitle={#1},%
    pdfauthor={#2},%
    pdfkeywords={#3}%
  }%
}
