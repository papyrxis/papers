\usepackage{etoolbox}

\def\addtocatlist#1#2{%
  \ifcsdef{Skill#1}{%
    \expandafter\appto\csname Skill#1\endcsname{, #2}%
  }{%
    \expandafter\gdef\csname Skill#1\endcsname{#2}%
  }%
}

% \newskill{Category}{DisplayLabel}{on|off}
\newcommand{\newskill}[3]{%
  \ifstrequal{#3}{on}{%
    \addtocatlist{#1}{#2}%
  }{}%
}
