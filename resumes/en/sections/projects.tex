% ── cdin ─────────────────────────────────────
\newcommand{\projcdin}{%
  \proj{cdin}{C $\cdot$ Lua $\cdot$ SDL3 $\cdot$ Text Editor}{%
    \item Open-source text editor with a C runtime (SDL3, stb\_truetype) and a fully
    Lua-based editor layer (documents, keymaps, plugins) loaded at runtime without
    recompilation. Supports Normal/Insert/Visual modes, Ex command line, project tree
    with Git markers, cell-based render cache, and cross-platform builds (Linux, macOS, Windows).%
  }{\ProjCdinShort}%
}
