% personal ─────────────────────────────────────────────────────────────────────
\newcommand{\pbApplyPersonal}{%
  \geometry{a4paper,top=3cm,bottom=3cm,left=3.5cm,right=3.5cm,
            headheight=1.2cm,headsep=0.8cm,footskip=1.2cm}
  \setstretch{1.35}
  \setlength{\parindent}{0pt}
  \setlength{\parskip}{0.8em}
  \titleformat{\section}
    {\large\bfseries\color{PBPrimary}}
    {\thesection.}{0.5em}{}[\vspace{0.2ex}{\color{PBBorder}\titlerule}\vspace{0.4ex}]
  \titleformat{\subsection}
    {\normalsize\bfseries\color{PBInk}}{\thesubsection}{0.5em}{}
  \titlespacing*{\section}{0pt}{2.5ex plus .5ex}{1.5ex plus .3ex}
  \titlespacing*{\subsection}{0pt}{1.8ex plus .4ex}{0.8ex plus .2ex}
}

% academic ─────────────────────────────────────────────────────────────────────
\newcommand{\pbApplyAcademic}{%
  \geometry{a4paper,top=2.5cm,bottom=2.5cm,left=2.5cm,right=2.5cm,
            headheight=1cm,headsep=0.7cm,footskip=1cm}
  \setstretch{1.2}
  \setlength{\parindent}{1.5em}
  \setlength{\parskip}{0.2ex}
  \titleformat{\section}
    {\normalfont\Large\bfseries}{\Roman{section}.}{0.5em}{\MakeUppercase}
  \titleformat{\subsection}
    {\normalfont\large\bfseries}{\Alph{subsection}.}{0.5em}{}
  \titleformat{\subsubsection}
    {\normalfont\normalsize\itshape}{\arabic{subsubsection})}{0.5em}{}
  \titlespacing*{\section}{0pt}{3ex plus .5ex}{1.5ex plus .3ex}
  \titlespacing*{\subsection}{0pt}{2ex plus .4ex}{1ex plus .2ex}
}

% ieee ─────────────────────────────────────────────────────────────────────────
\newcommand{\pbApplyIEEE}{%

  \setlength{\parindent}{1em}
  \setlength{\parskip}{0pt}
  \titleformat{\section}
    {\normalfont\large\bfseries\MakeUppercase}{\Roman{section}.}{0.5em}{}
  \titleformat{\subsection}
    {\normalfont\normalsize\itshape}{\Alph{subsection}.}{0.5em}{}
  \titleformat{\subsubsection}
    {\normalfont\normalsize\itshape}{\arabic{subsubsection})}{0.5em}{}
  \titlespacing*{\section}{0pt}{2.5ex plus .5ex}{1.5ex plus .3ex}
  \titlespacing*{\subsection}{0pt}{1.5ex plus .3ex}{0.5ex plus .2ex}
}

  \IfStrEqCase{\PaperStyle}{%
